CRIAR UM CADASTRO DE FROTAS DE VEICULOS

Um empresa (PESSOA) pode ter mais de um veículo. Escreva um programa que cadastra os veiculos de uma empresa. Caso a Empresa ainda nao exista, sua tarefa é armazenar a Empresa na tabela destinada a ela.


Exemplo de dado:
{
    "nome": "Localiza Aluguel de Carros",
    "documento": "85.728.082/0001-82",
    "telefone": "(11) 82343-9202",
    "site": "www.localiza.com",
    "veiculos": [
        {"placa": "GWR-2I38", "montadora": "FIAT", "modelo": "Pulse", "ano": 2023},
        {"placa": "HRT-0E65", "montadora": "Honda", "modelo": "HRV", "ano": 2024},
        {"placa": "CHK-5F72", "montadora": "GM", "modelo": "Onix", "ano": 2021}  
    ]
}

CREATE TABLE empresa(
    id number generated by default as identity,
    nome varchar(100) not null,
    site varchar(50),
    documento varchar(50),
    telefone varchar(50),
    primary key(id)
);

CREATE TABLE carro(
    id number generated by default as identity,
    montadora varchar(50) not null,
    modelo varchar(50) not null,
    ano number,
    placa varchar(20),
    proprietario_id number,
    primary key(id),
    foreign key(proprietario_id) references empresa(id)
);
